\newcounter{row}
\newcounter{col}

\newcommand\setrow[3]{
	\setcounter{col}{1}
	\foreach \n in {#1, #2, #3} {
		\edef\x{\value{col} - 0.5}
		\edef\y{5.5 - \value{row}}
		\node[anchor=center] at (\x, \y) {\n};
		\stepcounter{col}
	}
	\stepcounter{row}
}
